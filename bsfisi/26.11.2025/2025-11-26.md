Bezugspreiskalkulation:
- Listeneinkaufspreis (Netto Verkauspreis)
- Zieleinkaufspreis = Netto-Rabatte
- Bareinkaufspreis = Zieleinkaufspreis-Skonto
- Bezugspreis = Bareinkaufspreis+Bezugskosten/lieferkosten

```C++
const int var1 = D8;

void setup() {
  Serial.begin(115200);
  pinMode(var1, INPUT);
}

void loop() {
  if (digitalRead(var1)) {
    Serial.println("on");
  } else {
    Serial.println("off");
  }
  delay(200);
}
```
D8=TOUCH
D0=INTERNAL LED
D7=SPEAKER
```C++
int melody[] = {1000,2500};  
int noteDurations[] = {50, 50};

void setup() {
  Serial.begin(9600);
  pinMode(D8, INPUT);
  pinMode(D0, OUTPUT);
  pinMode(D7, OUTPUT);
}

void loop() {
  if (digitalRead(D8)) {
    Serial.println("on");
    digitalWrite(D0, LOW);

    for (int i = 0; i < 8; i++) {
      tone(D7, melody[i]);
      delay(noteDurations[i]);
      noTone(D7);
      delay(5);
    }

  } else {
    Serial.println("off");
    digitalWrite(D0, HIGH);
    noTone(D7);
  }
  delay(0);
}
```
